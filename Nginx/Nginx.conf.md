## 反向代理

```
server {
    listen 80;
    server_name github.com *.github.com;
    location / {
        proxy_pass         http://localhost:5010;
        proxy_http_version 1.1;
        proxy_set_header   Connection keep-alive;
        proxy_set_header   Host $host;
        proxy_set_header   Upgrade $http_upgrade;
        proxy_set_header   X-Real-IP $remote_addr;
        proxy_set_header   X-Forwarded-For $proxy_add_x_forwarded_for;
        proxy_set_header   X-Forwarded-Proto $scheme;
        proxy_cache_bypass $http_upgrade;
    }
}
```

## 普通网站配置

```
server {
       listen 80;
       listen [::]:80;

       server_name github.com *.github.com;

       access_log /var/log/nginx/github.com_access.log;
       error_log /var/log/nginx/github.com_error.log;

       root /var/www/github.com;
       index index index.html index.htm;

       location ~ .*\.(?:htm|html)$ {
              add_header Cache-Control "private, no-store, no-cache, must-revalidate, proxy-revalidate";
       }

       location ~ .*\.(js|css|jpg|png|gif)(.*) {
               expires 7d;
       }

       # Deny all attempts to access hidden files such as .htaccess, .htpasswd
       location ~ /\. {
                deny all;
                access_log off;
                log_not_found off;
       }

       location / {
               try_files $uri $uri/ =404;
       }
}
```

### 负载均衡

```
upstream  www.github.com   
{
    server   10.0.1.50:8080;
    server   10.0.1.51:8080;
}

server
{
    listen  80;
    server_name  www.github.com;

    location / {
        proxy_pass        http://www.github.com;
        proxy_set_header   Host             $host;
        proxy_set_header   X-Real-IP        $remote_addr;
        proxy_set_header   X-Forwarded-For  $proxy_add_x_forwarded_for;
    }
}
```
